T = (stable);
t1 = 0:T:0.2;
N = length(t1);

u = ones(4,N);
x = zeros(4,N);
Fx = zeros(4,N);
y = ones(1,N);

Fx(:,1) = A*x(:,1) + B*u(1);
x(:,2) = x(:,1) + (T*Fx(:,1));

Fx(:,2) = A*x(:,2) + B*u(2);
x(:,3) = x(:,2) + ((3/2)*T*Fx(:,2)) - ((1/2)*T*Fx(:,1));

for k=1:N-3
    Fx(:,k+2) = A*x(:,k+2) + B*u(k+2);
    x(:,k+3) = x(:,k+2) + ((23/12)*T*Fx(:,k+2)) - ((16/12)*T*Fx(:,k+1)) + ((5/12)*T*Fx(:,k));
    y1(k+3) = C*x(:,k+3);
end

figure()
subplot(2,1,1)
plot(t1,y1)
title('lamdaT for Stable and Accurate T')

T = (unstable);

N = length(t2);

u = ones(4,N);
x = zeros(4,N);
Fx = zeros(4,N);

for k=1:N-3
    Fx(:,k+2) = A*x(:,k+2) + B*u(k+2);
    x(:,k+3) = x(:,k+2) + ((23/12)*T*Fx(:,k+2)) - ((16/12)*T*Fx(:,k+1)) + ((5/12)*T*Fx(:,k));
